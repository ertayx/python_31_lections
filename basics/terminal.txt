директория = папка
/ - корневая папка (здесь находятся все конфигурации ОС)
~ - домашняя папка

ls - показывает все файлы и папки в текущей директории
ls {название папки} - показывает все файлы и папки внутри другой директории
ls -a - показывает все скртытые папки и файлы в текущей директории
ls -l - показывает более подробную информацию о файлах и папках


cd  - переходит в домашнюю папку
cd ~ - переходит в домашнюю папку
cd {название папки} - переходит в другую директорию
cd .. - возвращяется на папку выше (кнопка назад в windows)
cd . - переходит в текущую папку (остается в ней же)

mkdir - создает папку
rmdir - удаляет пустую папку
rm -rf - удаляет папку или файл со всем содержимым (безвозвратно)

sudo - права суперюзера (superuser do)

touch - создает файл
nano - открывает редактор кода
cat - показывает содержимое файла

pwd - показывает абсолютный путь(с начала корнвой папки /) до текущей директории

code - открывает vscode
code . - открывает vscode в текущей директории

history - показывает историю комманд

mv test test1 - перезаписывает или переименовывает файл либо папку
mv test.txt test2/ - перемещает файл test.txt в папку test2

echo "hello world" > test.txt - вставляет текст в файл
